    <script type="text/javascript">
        $(document).ready(function () {
            $(".block").click(function () {
                var type = $(this).attr("type");
                var db = $(this).attr("db");

                if (type == "disk")
                {
                    window.external.openDisk(db);
                }

                if (type == "folder")
                {
                    var id = $(this).attr("id");
                    window.external.openFolder(id, db);
                }

                if (type == "document")
                {
                    var id = $(this).attr("id");
                    window.external.openDocument(id, db);
                }
            });
        });
